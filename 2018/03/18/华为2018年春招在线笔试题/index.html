大数乘法

问题的描述很简单，就是实现两个带符号大数的乘法。具体的思路如下：
因为涉及到大数，所以肯定不能够使用常规的运算方法，即使使用 long long类型的数据还是会超出其表示范围。一种可行的方法就是使用字符串来表示一串数字，然后自己来通过字符操作来实现乘法的具体的过程：一个数的每一位分别乘以另外一个数，在乘的过程中注意移位的变化，最后再将所有的中间结果按照正确的位置相加得到最后的结果。其实就是我们将我们小学数学乘法的过程化，hahaha……但是这样算法的事件复杂度比较高，刚刚看网上有时间复杂度为O(lgn)的方法.

#include <iostream>                                                                                                                          
#include <string>

using namespace std;

string  Large_Multiplication(string str1, string str2) {
    int len1 = str1.length();
    int len2 = str2.length();
    /*
     *两个数A和B，A的位数为m，B的位数为n，
     *则乘积结果要么为 m+n-1位（最高位无进位）或者为 m+n位（最高位有进位）
     */
    string re(len1 + len2, '0');
    //使用第二个数的每一位乘以第一个数得到中间结果
    for(int i = len2 - 1;  i >= 0; i--) {
        int carry_pre = 0;  //每一位的乘法中向下一位的进位
        int carry_add = 0;  //计算中间和时向下一位的进位
        for(int j = len1 - 1; j >= 0; j--) {
            int num = (str2[i] - '0') * (str1[j] - '0') + carry_pre;
            carry_pre = num / 10;
            num %= 10;

            //计算竖向中间和，填充结果的i+j+1位
            int sum = (re[i+j+1] - '0') + num + carry_add;
            re[i+j+1] = sum % 10 + '0';
            carry_add = sum / 10;
        }
        re[i] += carry_add + carry_pre;  //处理进位
    }

    if(re[0] == '0') {  //判断乘积的位数
        re = re.substr(1);
    }
    return re;
}

int main() {
    string num1, num2;
    cin>>num1>>num2;
    if(num1[0] == '0' || num2[0] == '0') {
        cout<<0<<endl;
    }
    else {
        int sign1 = 1, sign2 = 1;  //记录两个数的正负号
        string a, b;   //a, b参数用于去符号乘法
        if(num1[0] == '-') {
            sign1 = -1;
            a = num1.substr(1);
        } else  {
            a = num1;
        }
        if(num2[0] == '-') {
            sign2 = -1;
            b = num2.substr(1);
        } else {
            b= num2;
        }

        string product = Large_Multiplication(a, b);

        if(sign1 * sign2 == -1) product = "-" + product;
        cout<<product<<endl;
    }
    return 0;
}

另外一种思路：上面一种思路主要是面上过程化，需要注意的细节更多。下面这种思路更过的是面向结果，不要考虑那么多细节，要更简单一点，传入的参数还是要先把符号处理了。

vector<int> Large_Multiplication(string num1, string num2) {
    int len1 = num1.length();
    int len2 = num2.length();
    vector<int> re (len1 + len2, 0);
    for(int i = len2 - 1; i >= 0; i--) {
        for(int j = len1 - 1; j >= 0; j--) {
            re[i + j + 1] += (num2[i] - '0') * (num1[j] - '0');
        }
    }
    int carry = 0;
    for(int i = re.size() - 1; i >= 0; i--) {
        int sum = re[i] + carry;
        carry = sum / 10;
        re[i] = sum % 10;
    }
    return re;
}
